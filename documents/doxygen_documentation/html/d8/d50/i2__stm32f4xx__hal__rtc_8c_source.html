95  * @brief Set alarm status.
96  * @details Set status of required RTC Alarm.
97  *
98  * @param[in] *handle RTC Handle.
99  * @param[in] *alarm_id RTC alarm instance to read.
100  * @return Status of alarm setup.

102 static bool rtc_alarm_status_get(RTC_HandleTypeDef *handle, rtc_alarm_t alarm_id)
103 {

105

115 /**
116  * @brief RTC initialization.
117  * @details Initialize RTC peripherals and respective clocks.
118  *
119  * @return None.
120  */
121 i2_error i2_rtc_init(void)
122 {
123  i2_error retval = I2_SUCCESS;
124  RTC_HandleTypeDef *handle;
125  RTC_TypeDef *base;
126  int32_t i;
127 
128  /* Prevent re-init */
129  if ( ctx.initialized ) {
130  return I2_NOT_AVAILABLE;
131  }
132 
133  /* Get the pointer to the RTC handle */
134  handle = &(ctx.rtc);
135 
136  /* Get the RTC controller base address */
137  base = ctx.baseaddr;
138 
139 #if defined ( ENABLE_RTOS_AWARE_HAL )
140  /* Create mutex. */
141  ctx.mutex = xSemaphoreCreateMutex();
142 #endif
143 
144  if ( base == RTC ) {
145  RCC_PeriphCLKInitTypeDef PeriphClkInitStruct;
146 
147  HAL_PWR_EnableBkUpAccess();
148 
149  PeriphClkInitStruct.PeriphClockSelection = RCC_PERIPHCLK_RTC;
150  if ( i2_is_lse_on() ) {
151  PeriphClkInitStruct.RTCClockSelection = RCC_RTCCLKSOURCE_LSE;
152  } else {
153  PeriphClkInitStruct.RTCClockSelection = RCC_RTCCLKSOURCE_LSI;
154  }
155 
156  if (HAL_RCCEx_PeriphCLKConfig(&PeriphClkInitStruct) != HAL_OK) {
157  return I2_FAILURE;
158  }
159 
160  __HAL_RCC_RTC_ENABLE();
161  rtc = handle;
162  } else {
163  return I2_INVALID_PARAM;
164  }
165 
166  HAL_NVIC_SetPriority(RTC_Alarm_IRQn,
167  RTC_PREEMPTION_PRIORITY,
168  RTC_SUB_PRIORITY);
169  HAL_NVIC_EnableIRQ(RTC_Alarm_IRQn);
170 
171  HAL_NVIC_SetPriority(RTC_WKUP_IRQn, RTC_PREEMPTION_PRIORITY, RTC_SUB_PRIORITY);
172  HAL_NVIC_EnableIRQ(RTC_WKUP_IRQn);
173 
174  handle->Instance = base;
175  handle->Init.HourFormat = RTC_HOURFORMAT_24;
176  if ( i2_is_lse_on() ) {
177  handle->Init.AsynchPrediv = 127;
178  handle->Init.SynchPrediv = 255;
179  } else {
180  handle->Init.AsynchPrediv = RTC_ASYNCH_PREDIV;
181  handle->Init.SynchPrediv = RTC_SYNCH_PREDIV;
182  }
183  handle->Init.OutPut = RTC_OUTPUT_DISABLE;
184  handle->Init.OutPutPolarity = RTC_OUTPUT_POLARITY_HIGH;
185  handle->Init.OutPutType = RTC_OUTPUT_TYPE_OPENDRAIN;
186 
187  /* Initialize RTC with default sync prediv value */
188  if (HAL_RTC_Init(handle) != HAL_OK) {
189  return I2_FAILURE;
190  }
191 
192  for (i = 0; i < MAX_NUM_RTC_ALARMS; i++) {
193  ctx.alarm[i].cb = NULL;
194  ctx.alarm[i].arg = NULL;
195  }
196  ctx.wakeup.cb = NULL;
197  ctx.wakeup.arg = NULL;
198 
199  ctx.initialized = true;
200 
201  return retval;
202 }

248  retval = I2_FAILURE;

256 err:

328 /**
329  * @brief Get RTC alarm.
330  * @details Read alarm time and repetition settings.
331  *
332  * @param[in] *alarm_id Alarm ID to read.
333  * @param[out] *alarm_type Gets updated with alarm type @ref rtc_alarm_type_t.
334  * @param[out] *alarm Gets updated with time setting for alarm.
335  * @return Error code @ref I2_ERROR.
336  *
337  * @note If the given alarm ID is not armed, the alarm_type will be set
338  * to RTC_ALARM_TYPE_NONE and the alarm data meaningless.
339  */
340 i2_error i2_rtc_alarm_get(rtc_alarm_t alarm_id, rtc_alarm_type_t *alarm_type,
341  struct tm *alarm)
342 {
343  i2_error retval = I2_SUCCESS;
344  RTC_HandleTypeDef *handle;
345  RTC_AlarmTypeDef _alarm;
346 
347  if ( !ctx.initialized || !alarm || !alarm_type ) {
348  return I2_INVALID_PARAM;
349  }
350 
351  if ( (alarm_id < 0) || (alarm_id >= MAX_NUM_RTC_ALARMS) ) {
352  return I2_INVALID_PARAM;
353  }
354 
355 #if defined ( ENABLE_RTOS_AWARE_HAL )
356  xSemaphoreTake(ctx.mutex, (TickType_t)portMAX_DELAY);
357 #endif
358 
359  /* Get the pointer to the RTC handle */
360  handle = &(ctx.rtc);
361 
362  /* Choose the alarm A or B */
363  if (alarm_id == RTC_ALARM_1) {
364  _alarm.Alarm = RTC_ALARM_A;
365  } else {
366  _alarm.Alarm = RTC_ALARM_B;
367  }
368 
369  if (HAL_RTC_GetAlarm(handle, &_alarm, _alarm.Alarm, RTC_FORMAT_BIN) != HAL_OK) {
370  retval = I2_FAILURE;
371  goto out;
372  }
373 
374  /* Get the hour, min, and sec */
375  alarm->tm_hour = _alarm.AlarmTime.Hours;
376  alarm->tm_min = _alarm.AlarmTime.Minutes;
377  alarm->tm_sec = _alarm.AlarmTime.Seconds;
378  alarm->tm_isdst = -1;
379 
380  /* RTC alarm does not match year or month */
381  alarm->tm_year = 0;
382  alarm->tm_mon = 0;
383 
384  /* Check if it has been armed prior. If not, nothing is setup */
385  if (!rtc_alarm_status_get(handle, alarm_id)) {
386  *alarm_type = RTC_ALARM_TYPE_NONE;
387  } else if (_alarm.AlarmDateWeekDaySel == RTC_ALARMDATEWEEKDAYSEL_WEEKDAY) {
388  /* Alarm based on the week day */
389  *alarm_type = RTC_ALARM_TYPE_WEEKDAY;
390  alarm->tm_wday = (_alarm.AlarmDateWeekDay == 7) ? 0 : _alarm.AlarmDateWeekDay;
391  } else if (_alarm.AlarmDateWeekDaySel == RTC_ALARMDATEWEEKDAYSEL_DATE) {
392  /* Alarm based on date */
393  *alarm_type = RTC_ALARM_TYPE_DATE;
394  alarm->tm_mday = _alarm.AlarmDateWeekDay;
395  }
396 
397 out:
398 #if defined ( ENABLE_RTOS_AWARE_HAL )
399  xSemaphoreGive(ctx.mutex);
400 #endif
401 
402  return retval;
403 }

405 /**
406  * @brief Set RTC alarm.
407  * @details Configure alarm with time and repetition settings.
408  *
409  * @param[in] alarm_id Alarm ID to read.
410  * @param[in] alarm_type Alarm type to update @ref rtc_alarm_type_t.
411  * @param[in] *alarm Time setting for alarm to update.
412  * @return Error code @ref I2_ERROR.
413  *
414  * @note If the given alarm_id is already set/armed, it will overwrite
415  * the existing alarm setting without any error.
416  */
417 i2_error i2_rtc_alarm_set(rtc_alarm_t alarm_id, rtc_alarm_type_t alarm_type,
418  struct tm *alarm)
419 {
420  i2_error retval = I2_SUCCESS;
421  RTC_HandleTypeDef *handle;
422  RTC_AlarmTypeDef _alarm;
423 
424  if (!ctx.initialized || !alarm) {
425  return I2_INVALID_PARAM;
426  }
427 
428  if ( (alarm_id < 0) || (alarm_id >= MAX_NUM_RTC_ALARMS) ) {
429  return I2_INVALID_PARAM;
430  }
431 
432 #if defined ( ENABLE_RTOS_AWARE_HAL )
433  xSemaphoreTake(ctx.mutex, (TickType_t)portMAX_DELAY);
434 #endif
435 
436  /* Set the alarm_type to a valid value */
437  if ((alarm_type != RTC_ALARM_TYPE_WEEKDAY) &&
438  (alarm_type != RTC_ALARM_TYPE_DATE)) {
439  retval = I2_INVALID_PARAM;
440  goto out;
441  }
442 
443  /* Get the pointer to the RTC handle */
444  handle = &(ctx.rtc);
445 
446  /* Choose the alarm A or B */
447  if ( alarm_id == RTC_ALARM_1 ) {
448  _alarm.Alarm = RTC_ALARM_A;
449  } else {
450  _alarm.Alarm = RTC_ALARM_B;
451  }
452  /* Do not mask out anything. We will match day/date, hour, min, and sec */
453  _alarm.AlarmMask = RTC_ALARMMASK_NONE;
454 
455  /* Depending on alarm type, we can match day or date */
456  if (alarm_type == RTC_ALARM_TYPE_WEEKDAY) {
457  _alarm.AlarmDateWeekDaySel = RTC_ALARMDATEWEEKDAYSEL_WEEKDAY;
458  /* Set the weekday */
459  _alarm.AlarmDateWeekDay = (!alarm->tm_wday ? 7 : alarm->tm_wday);
460  } else if (alarm_type == RTC_ALARM_TYPE_DATE) {
461  _alarm.AlarmDateWeekDaySel = RTC_ALARMDATEWEEKDAYSEL_DATE;
462  /* Set date */
463  _alarm.AlarmDateWeekDay = alarm->tm_mday;
464  } else {
465  retval = I2_INVALID_PARAM;
466  goto out;
467  }
468 
469  /* Don't care about matching the alarm at sub-second level */
470  _alarm.AlarmSubSecondMask = RTC_ALARMSUBSECONDMASK_ALL;
471 
472  /* Set the hour, min, and sec */
473  _alarm.AlarmTime.TimeFormat = (alarm->tm_hour < 12) ? RTC_HOURFORMAT12_AM : RTC_HOURFORMAT12_PM;
474  _alarm.AlarmTime.Hours = alarm->tm_hour;
475  _alarm.AlarmTime.Minutes = alarm->tm_min;
476  _alarm.AlarmTime.Seconds = alarm->tm_sec;
477  _alarm.AlarmTime.SubSeconds = 0;
478 
479  if (HAL_RTC_SetAlarm_IT(handle, &_alarm, RTC_FORMAT_BIN) != HAL_OK) {
480  retval = I2_FAILURE;
481  } else {
482  rtc_alarm_status_set(handle, alarm_id, true);
483  }
484 
485 out:
486 #if defined ( ENABLE_RTOS_AWARE_HAL )
487  xSemaphoreGive(ctx.mutex);
488 #endif
489 
490  return retval;
491 }
